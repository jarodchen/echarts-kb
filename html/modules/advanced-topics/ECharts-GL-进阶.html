ECharts-GL 概述 ​

什么是ECharts-GL? ​

ECharts-GL是ECharts的3D扩展库,基于WebGL技术,提供:

  • 3D图表: bar3D、scatter3D、surface、line3D
  • 地理可视化: globe(地球仪)、map3D
  • 点云处理: 大规模3D散点图
  • 体渲染: 医学影像、气象数据

// 3D曲面 - 适合数学函数可视化
function generateSurfaceData(): number[][] {
  const data: number[][] = [];
  
  for (let x = -10; x <= 10; x += 0.5) {
    for (let y = -10; y <= 10; y += 0.5) {
      // z = sin(sqrt(x² + y²)) / sqrt(x² + y²)
      const r = Math.sqrt(x * x + y * y);
      const z = Math.sin(r) / (r || 1);
      data.push([x, y, z]);
    }
  }
  
  return data;
}

chart.setOption({

      main: {
        intensity: 2,
        shadow: true
      },

      }

WebGL渲染原理 ​

渲染管线流程 ​

顶点着色器(Vertex Shader) ​

glsl
// 简化的顶点着色器
attribute vec3 aPosition;   // 顶点位置
attribute vec3 aNormal;     // 法向量
attribute vec2 aTexCoord;   // 纹理坐标

uniform mat4 uModelMatrix;      // 模型矩阵
uniform mat4 uViewMatrix;       // 视图矩阵
uniform mat4 uProjectionMatrix; // 投影矩阵

varying vec3 vNormal;       // 传递给片元着色器
varying vec2 vTexCoord;

void main() {
  // 转换到裁剪空间
  gl_Position = uProjectionMatrix * uViewMatrix * uModelMatrix * vec4(aPosition, 1.0);
  
  // 传递法向量和纹理坐标
  vNormal = aNormal;
  vTexCoord = aTexCoord;
}

片元着色器(Fragment Shader) ​

glsl
// Lambert光照模型的片元着色器
precision mediump float;

varying vec3 vNormal;
varying vec2 vTexCoord;

uniform vec3 uLightDirection;  // 光源方向
uniform vec3 uColor;           // 物体颜色

void main() {
  // 计算光照强度
  float intensity = dot(normalize(vNormal), normalize(uLightDirection));
  intensity = max(intensity, 0.0);  // 确保非负
  
  // 最终颜色 = 物体颜色 × 光照强度
  gl_FragColor = vec4(uColor * intensity, 1.0);
}

自定义Shader ​

为什么需要自定义Shader? ​

ECharts-GL内置的着色器可能无法满足以下需求:

  1. 特殊效果: holographic(全息)、x-ray(透视)
  2. 数据编码: 用颜色表示第4/5维数据
  3. 后处理: bloom(泛光)、DOF(景深)
  4. 性能优化: 简化计算,提升FPS

注册自定义着色器 ​

typescript
import { registerShader } from 'echarts-gl';

registerShader('customHolographic', {
  uniforms: {
    uTime: { type: 'number', value: 0 },
    uOpacity: { type: 'number', value: 0.6 }
  },
  
  vertexShader: `
    attribute vec3 position;
    attribute vec3 normal;
    
    uniform mat4 modelViewMatrix;
    uniform mat4 projectionMatrix;
    
    varying vec3 vNormal;
    varying vec3 vPosition;
    
    void main() {
      vNormal = normal;
      vPosition = position;
      
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  
  fragmentShader: `
    precision mediump float;
    
    varying vec3 vNormal;
    varying vec3 vPosition;
    
    uniform float uTime;
    uniform float uOpacity;
    
    void main() {
      // 全息扫描效果
      float scan = sin(vPosition.y * 10.0 - uTime * 2.0) * 0.5 + 0.5;
      
      // 边缘光
      float fresnel = pow(1.0 - abs(dot(vNormal, vec3(0.0, 0.0, 1.0))), 2.0);
      
      // 组合颜色
      vec3 color = vec3(0.0, 1.0, 1.0) * scan + vec3(0.5, 0.0, 1.0) * fresnel;
      
      gl_FragColor = vec4(color, uOpacity);
    }
  `
});

// 使用自定义着色器
chart.setOption({
  series: [{
    type: 'bar3D',
    data: data,
    shading: 'customHolographic',  // ✅ 使用自定义着色器
    itemStyle: {
      opacity: 0.6
    }
  }]
});

// 动画更新uniform
setInterval(() => {
  const shader = echarts.gl.shaderManager.getShader('customHolographic');
  shader.uniforms.uTime.value += 0.016;  // 更新时间
  chart.setOption({});  // 触发重绘
}, 16);

实例: 热力图自定义着色器 ​

typescript
// 用颜色表示温度+高度的多维数据
registerShader('temperatureHeatmap', {
  uniforms: {
    uMinValue: { type: 'number', value: 0 },
    uMaxValue: { type: 'number', value: 100 }
  },
  
  vertexShader: `
    attribute vec3 position;
    attribute float value;  // 温度值
    
    uniform mat4 modelViewMatrix;
    uniform mat4 projectionMatrix;
    
    varying float vValue;
    
    void main() {
      vValue = value;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  
  fragmentShader: `
    precision mediump float;
    
    varying float vValue;
    
    uniform float uMinValue;
    uniform float uMaxValue;
    
    // 温度颜色渐变: 蓝→青→绿→黄→红
    vec3 temperatureColor(float t) {
      t = clamp(t, 0.0, 1.0);
      
      if (t < 0.25) {
        return mix(vec3(0.0, 0.0, 1.0), vec3(0.0, 1.0, 1.0), t / 0.25);
      } else if (t < 0.5) {
        return mix(vec3(0.0, 1.0, 1.0), vec3(0.0, 1.0, 0.0), (t - 0.25) / 0.25);
      } else if (t < 0.75) {
        return mix(vec3(0.0, 1.0, 0.0), vec3(1.0, 1.0, 0.0), (t - 0.5) / 0.25);
      } else {
        return mix(vec3(1.0, 1.0, 0.0), vec3(1.0, 0.0, 0.0), (t - 0.75) / 0.25);
      }
    }
    
    void main() {
      float normalizedValue = (vValue - uMinValue) / (uMaxValue - uMinValue);
      vec3 color = temperatureColor(normalizedValue);
      
      gl_FragColor = vec4(color, 1.0);
    }
  `
});

GPU内存管理 ​

typescript
class GLResourceManager {
  private chart: echarts.ECharts;
  
  constructor(chart: echarts.ECharts) {
    this.chart = chart;
  }
  
  /**
   * 清理GPU资源
   */
  dispose() {
    // 清理ECharts-GL资源
    if (this.chart.gl) {
      this.chart.gl.dispose();
    }
    
    // 清理ECharts资源
    this.chart.dispose();
  }
  
  /**
   * 监控显存使用
   */
  monitorVRAM() {
    const gl = this.chart.gl.getContext();
    
    // WebGL扩展
    const ext = gl.getExtension('WEBGL_debug_renderer_info');
    if (ext) {
      const vendor = gl.getParameter(ext.UNMASKED_VENDOR_WEBGL);
      const renderer = gl.getParameter(ext.UNMASKED_RENDERER_WEBGL);
      
      console.log('GPU:', vendor, renderer);
    }
  }
}

// 使用
const manager = new GLResourceManager(chart);

// 组件卸载时
onUnmounted(() => {
  manager.dispose();
});

// 根据距离动态调整细节

实战案例 ​

案例1: 全球人口分布3D可视化 ​

typescript
class GlobalPopulation3D {
  private chart: echarts.ECharts;
  
  constructor() {
    this.chart = echarts.init(document.getElementById('globe'));
  }
  
  async init() {
    // 1. 加载世界地图GeoJSON
    const worldGeoJSON = await fetch('/geojson/world.json').then(r => r.json());
    echarts.registerMap('world', worldGeoJSON);
    
    // 2. 加载人口数据
    const populationData = await fetch('/api/population').then(r => r.json());
    
    // 3. 转换为3D散点 [经度, 纬度, 人口数]
    const scatterData = populationData.map(city => [
      city.lng,
      city.lat,
      Math.log(city.population)  // 对数缩放
    ]);
    
    // 4. 渲染3D地球
    this.chart.setOption({
      globe: {
        baseTexture: '/textures/earth_day.jpg',
        heightTexture: '/textures/earth_topo.jpg',
        displacementScale: 0.05,
        shading: 'lambert',
        environment: '/textures/starfield.jpg',
        
        viewControl: {
          autoRotate: true,
          autoRotateSpeed: 5
        },
        
        light: {
          main: {
            intensity: 2,
            shadow: true
          },
          ambient: {
            intensity: 0.3
          }
        }
      },
      series: [{
        type: 'scatter3D',
        coordinateSystem: 'globe',
        data: scatterData,
        symbolSize: (params) => params[2] * 2,
        
        itemStyle: {
          color: (params) => {
            const population = Math.exp(params[2]);
            if (population > 10000000) return '#ff0000';  // 超大城市
            if (population > 1000000) return '#ff6600';   // 大城市
            return '#00ff00';                             // 中小城市
          }
        },
        
        blendMode: 'lighter'  // 叠加混合
      }]
    });
  }
}

// 启动
const globe = new GlobalPopulation3D();
globe.init();

typescript
class MoleculeViewer {
  private chart: echarts.ECharts;
  
  constructor() {
    this.chart = echarts.init(document.getElementById('molecule'));
  }
  
  renderMolecule(atoms: any[], bonds: any[]) {
    // 原子类型颜色映射
    const atomColors: Record<string, string> = {
      'C': '#333333',  // 碳 - 黑色
      'H': '#ffffff',  // 氢 - 白色
      'O': '#ff0000',  // 氧 - 红色
      'N': '#0000ff'   // 氮 - 蓝色
    };
    
    this.chart.setOption({
      grid3D: {
        viewControl: {
          projection: 'orthographic'  // 正交投影(科学可视化常用)
        }
      },
      series: [
        // 化学键(线条)
        {
          type: 'line3D',
          data: bonds.map(bond => [
            atoms[bond.from].position,
            atoms[bond.to].position
          ]),
          lineStyle: {
            width: 4,
            color: '#888888'
          }
        },
        
        // 原子(球体)
        {
          type: 'scatter3D',
          data: atoms.map(atom => [
            atom.position[0],
            atom.position[1],
            atom.position[2],
            atom.type
          ]),
          symbolSize: (params) => {
            const sizes: Record<string, number> = {
              'C': 30, 'H': 20, 'O': 28, 'N': 29
            };
            return sizes[params[3]] || 25;
          },
          itemStyle: {
            color: (params) => atomColors[params[3]] || '#cccccc'
          },
          
          // 真实感渲染
          shading: 'realistic',
          realisticMaterial: {
            roughness: 0.3,
            metalness: 0.2
          }
        }
      ]
    });
  }
}

// 使用: 渲染水分子H₂O
const viewer = new MoleculeViewer();
viewer.renderMolecule(
  [
    { type: 'O', position: [0, 0, 0] },
    { type: 'H', position: [0.96, 0, 0] },
    { type: 'H', position: [-0.24, 0.93, 0] }
  ],
  [
    { from: 0, to: 1 },  // O-H键
    { from: 0, to: 2 }   // O-H键
  ]
);

📊 性能优化清单 ​

  • [ ] 超过1000点开启large: true
  • [ ] 使用progressive渐进式渲染
  • [ ] 简化symbolSize(小于8px)
  • [ ] 关闭不必要的交互(silent: true)
